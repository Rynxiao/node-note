## 操作系统

// \n
console.log(os.EOL);

//----------------------------------------------------------------------------

//

//              'LE' 小端模式
//----------------------------------------------------------------------------

// LE
console.log(os.endianness());

//          os.freemem() 方法以整数的形式回空闲系统内存 的字节数.

//          os.homedir() 方法以字符串的形式返回当前用户的home目录.
//----------------------------------------------------------------------------

//----------------------------------------------------------------------------

// RyndeMacBook-Pro.local

//          平均负载是UNIX相关的概念,在Windows平台上没有对应的概念. 在Windows上,其返回值总是[0, 0, 0].
//----------------------------------------------------------------------------

// [ 1.83203125, 1.7666015625, 1.75146484375 ]
